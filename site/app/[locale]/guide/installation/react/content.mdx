<StepSection>
    <Step $row>
        <StepL>
            ### <StepNum />Create a project
            If you don't have a React project, create one first. It's recommended to refer to [Start a New React Project](https://react.dev/learn/start-a-new-react-project).
        </StepL>
        <StepR>
            ```bash name=Terminal
            npm create vite@latest my-app -- --template react-ts
            cd my-app
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Install Master CSS
            Install `@master/css.vite` plugin into your project.
        </StepL>
        <StepR>
            ```bash action=install
            @master/css.vite@rc
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Create a configuration file
            Run the command to create a [master.css.js](/guide/configuration) file.
        </StepL>
        <StepR>
            ```bash action=create
            @master/css@rc
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Set up Master CSS
            Add a `masterCSS` vite plugin in `vite.config.js`.

            - By default, `options.mode` is set to `runtime`.
        </StepL>
        <StepR>
            ```js name=vite.config.js
            import { defineConfig } from 'vite'
            import masterCSS from '@master/css.vite' // [!code highlight]

            export default defineConfig({
                plugins: [
                    masterCSS() // [!code highlight]
                ]
            })
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Start dev server
            Run the command to start the Vite dev server.
        </StepL>
        <StepR>
            ```bash name=Terminal&action=run
            dev
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Start using Master CSS
            Now style your first element using Master CSS syntax!
        </StepL>
        <StepR>
            ```tsx name=src/App.tsx
            export default function App() {
                return (
                    <h1 className="italic m:12x fg:strong font:40 font:heavy">
                        Hello World
                    </h1>
                )
            }
            ```
        </StepR>
    </Step>
    <HelloWorld url="localhost:5173" />
</StepSection>
